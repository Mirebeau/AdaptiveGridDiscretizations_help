 80	def GetValue(self,key,default="_None",verbosity=2,array_float=False,
 81		help="Sorry : no help for this key",aggressive_memory_optimization_discard=False):
 82		"""
 83		Get a value from a dictionnary, printing some help if requested.
 84		"""
 85		# We only import arguments once, otherwise risks of issues with multiple defaults
 86		# Also, this allows to delete arguments and potentially save space
 87		assert key not in self.hfmOut['keys']['help']
 88
 89		self.hfmOut['keys']['help'][key] = help
 90		self.hfmOut['keys']['default'][key] = default
 91
 92		if key in self.hfmIn:
 93			self.hfmOut['keys']['used'].append(key)
 94			value = self.hfmIn[key]
 95			if aggressive_memory_optimization_discard and self.aggressive_memory_optimization:
 96				self.hfmIn[key] = (None,"Deleted for aggressive memory optimization")
 97			if self.clear_hfmIn:
 98				if isinstance(value,Metrics.Base) or ad.isndarray(value) and value.size>100:
 99					self.hfmIn.store[key] = (None,"Deleted in GetValue")
100			if array_float is False: return value
101			if not ad.is_ad(value): value = self.caster(value)
102			# Check shape
103			if isinstance(array_float,tuple):
104				shapeRef,shape = array_float,value.shape
105				if len(shapeRef)!=len(shape):
106					raise ValueError(f"Field {key} has incorrect number of dimensions. "
107						f"Expected shape {shapeRef}, found {shape}")
108				for sRef,s in zip(shapeRef,shape):
109					if sRef not in (-1,s): 
110						raise ValueError(f"Field {key} has incorrect dimensions. "
111							f"Expected shape {shapeRef}, found {shape}")
112			return value
113		elif isinstance(default,str) and default == "_None":
114			raise ValueError(f"Missing value for key {key}")
115		else:
116			if verbosity<=self.verbosity:
117				if isinstance(default,str) and default=="_Dummy":
118					print(f"see out['keys']['default'][{key}] for default")
119				else:print(f"key {key} defaults to {default}")
120			if isinstance(default, numbers.Number) and array_float is not False:
121				default = self.caster(default)
122			return default

202	def CostMetric(self,x):
203		"""Adimensionized interpolated metric""" 
204		if self._CostMetric is None:
205			self._CostMetric = self.metric.with_cost(self.cost)
206			# TODO : remove. No need to create this grid for our interpolation

208				for s in self.shape), indexing='ij')) # Adimensionized coordinates
209			self._CostMetric.set_interpolation(grid,periodic=self.periodic) # First order interpolation
210			if self._CostMetric_delete_dual: self._metric = (None,"Deleted in CostMetric")
211		return self._CostMetric.at(x)

221	def as_field(self,e,name,depth=0):
222		shape = self.hfmIn.shape
223		oshape,ishape = e.shape[:depth],e.shape[depth:]
224		if ishape==shape: # Already a field
225			return e
226		elif ishape==tuple(): # Constant field
227			return np.broadcast_to(e.reshape(oshape+(1,)*self.ndim),oshape+shape)
228		elif self.isCurvature:
229			if ishape==shape[self.ndim_phys:]:  # Angular field
230				return np.broadcast_to(e.reshape(oshape+(1,)*self.ndim_phys+ishape),oshape+shape)
231			elif ishape==shape[:self.ndim_phys]: # Physical field
232				return np.broadcast_to(e.reshape(oshape+ishape+(1,)*self.ndim_ang), oshape+shape)
233		raise ValueError(f"Field {name} has incorrect dimensions. Found {e.shape}, "
234			f"whereas domain has shape {shape}")

222	def as_field(self,e,name,depth=0):

224		oshape,ishape = e.shape[:depth],e.shape[depth:]

233				return np.broadcast_to(e.reshape(oshape+ishape+(1,)*self.ndim_ang), oshape+shape)
